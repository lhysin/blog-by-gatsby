* template syntax, rendering, binding, event handling

```html
<template>
  <!-- v-bind의 약어는 생략-->
  <a
    v-bind:class="rowClass"
    :href="link"
  >
    {{ message }}
  </a>
  <p>원시HTML {{ message }}</p>
  <div v-html="rawHtml"></div>

  <!-- 데이터 타입정의 -->
  <input v-model.number="obj.num" :disable="true" />

  <!-- enable javascript syntax -->
  <p>{{ 1 + 2 }}</p>
  <p>{{ '몇글자일까?'.substr(0, 2) }}</p>
  <p>{{ 1 === 1 ? 정답 : 오답 }}</p>

  <!-- event handling -->
  <button v-on:click="alert('클릭성공')" >
  <!-- custom methods로 이벤트 전파 -->
  <button :click="$ourAlert('클릭성공', $event)" >
  <!-- v-on의 약어 @ -->
  <button @click.prevent="alert('event prevent 후 클릭성공')" >

  <!-- 조건부 렌더링 v-if는 DOM에 미존재 -->
  <div
    v-if="false"
  >미존재</div>
  <!-- 조건부 렌더링 v-show는 style="display:none; -->
  <div
    v-show="false"
  >존재</div>

</template>
<script>
export default {
  name: 'app',
  /* 데이터 변경 리스닝 */
  watch : {
    'obj.num' : function(newVal, oldVal){
      alert(oldVal + ' => ' + newVal);
    }
  }
  data () {
    return {
      dynamicText: '동적Text',
      rawHtml: '<p>HTML입니다.</p>'
      rowClass: 'row',
      link: 'http://localhost',
      obj : {
        num: 23
      }
    }
  },
  methods: {
    /* 퍼블릭한 펑션엔 prefix로 '$' */
    $ourAlert : function $ourAlert(msg, event){
      console.log(event);
      alert(msg);
    }
  }
}
</script>
```

* v-for Array 타입
* 2.2.0 이상에서 v-for는 key 가 필수

```html
<template>

  <tbody>
    <!-- tr이 반복 -->
    <tr
      v-for="(obj, index) of list"
      :key="index"
    >
      <td>{{ obj.num }}</td>
    </tr>
    </template>
  </tbody>

  <tbody>
    <!-- template으로 감싼 영역이 반복, template은 소멸-->
    <template
      v-for="(obj, index) of list"
      :key="index"
    >
      <tr><td>{{ obj.num }}<</td></tr>
      <tr><td>{{ obj.num + 1 }}<</td></tr>
    </template>
  </tbody>

</template>
<script>
export default {
  name: 'app',
  data () {
    return {
      list: [{num : 1}, {num : 3}, {num : 5}, {num : 7}],
    }
  }
}
</script>
```
